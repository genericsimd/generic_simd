02367     //return vec_rsqrte(v);
02368     //Get the square root reciprocal estimate
02369     __vector float zero = (__vector float){0,0,0,0};
02370     __vector float oneHalf = (__vector float){0.5,0.5,0.5,0.5};
02371     __vector float one = (__vector float){1.0,1.0,1.0,1.0};
02372     __vector float estimate = vec_rsqrte( v.v );
02373     //One round of Newton-Raphson refinement
02374     __vector float estimateSquared = vec_madd( estimate, estimate, zero );
02375     __vector float halfEstimate = vec_madd( estimate, oneHalf, zero );
02376     __vector float r = vec_madd( vec_nmsub( v.v, estimateSquared, one ), halfEstimate, estimate );
02377     return _svec4_f(r);
02378 
02379 }

02901 FORCEINLINE _svec4_d svec_preduce_add(_svec4_d v0, _svec4_d v1, _svec4_d v2, _svec4_d v3) {
02902   //parallel reduction using mergeh mergel
02903   __vector double sv0 = v0.v[0] + v0.v[1];
02904   __vector double sv1 = v1.v[0] + v1.v[1];
02905   __vector double sv2 = v2.v[0] + v2.v[1];
02906   __vector double sv3 = v3.v[0] + v3.v[1];
02907 
02908   __vector double h0 = vec_mergeh(sv0, sv1);
02909   __vector double l0 = vec_mergel(sv0, sv1);
02910   __vector double h1 = vec_mergeh(sv2, sv3);
02911   __vector double l1 = vec_mergel(sv2, sv3);
02912 
02913   //reduction again
02914   __vector double s0 = h0 + l0;
02915   __vector double s1 = h1 + l1;
02916   return _svec4_d(s0, s1);
02917 }
